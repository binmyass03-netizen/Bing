        return (pa - pb).Magnitude
    end

    -- ── Safe FindFirstChild (nil parent = nil result) ──
    function lib.find(parent, name)
        if not parent then return nil end
        local ok, child = pcall(parent.FindFirstChild, parent, name)
        return (ok and child) and child or nil
    end

    -- ── Safe FindFirstChildOfClass ──
    function lib.findClass(parent, className)
        if not parent then return nil end
        local ok, child = pcall(parent.FindFirstChildOfClass, parent, className)
        return (ok and child) and child or nil
    end

    -- ── Safe tostring (prevents null-string crash from tostring on destroyed instances) ──
    function lib.str(v)
        if v == nil then return "nil" end
        local ok, s = pcall(tostring, v)
        return ok and s or "<invalid>"
    end

    -- ── Safe string concat for display names (crash RDX=0x0 at cmp [rdx+0x18]) ──
    function lib.displayName(player)
        if not player then return "Unknown" end
        local ok, dn = pcall(function() return player.DisplayName end)
        if not ok or not dn or dn == "" then
            local ok2, n = pcall(function() return player.Name end)
            return (ok2 and n and n ~= "") and n or "Unknown"
        end
        return dn
    end

    -- ── Iterate all players safely, skip LocalPlayer, skip nil characters ──
    -- callback(player, character, hrp) — only called when all 3 are valid
    function lib.forEachPlayer(callback, includeLocal)
        for _, player in Players:GetPlayers() do
            if not includeLocal and player == LP then continue end
            local char = player.Character
            if not char or not char.Parent then continue end
            local hrp = char:FindFirstChild("HumanoidRootPart")
            if hrp and hrp.Parent then
                local ok, err = pcall(callback, player, char, hrp)
            end
        end
    end

    -- ── Find closest other player within range ──
    function lib.closestPlayer(position, maxRange)
        maxRange = maxRange or 100
        local closest, bestDist = nil, maxRange
        lib.forEachPlayer(function(player, char, hrp)
            local d = (hrp.Position - position).Magnitude
            if d < bestDist then
                closest = player
                bestDist = d
            end
        end)
        return closest, bestDist
    end

    -- ── Find tool equipped by a player (safe) ──
    function lib.getTool(player)
        local char = lib.getChar(player)
        if not char then return nil end
        return char:FindFirstChildOfClass("Tool")
    end

    -- ── Protected Heartbeat wrapper (crash in callback won't kill the connection) ──
    function lib.safeHeartbeat(fn)
        return RunService.Heartbeat:Connect(function(dt)
            local ok, err = pcall(fn, dt)
        end)
    end

    -- ── Protected RenderStepped wrapper ──
    function lib.safeRenderStep(fn)
        return RunService.RenderStepped:Connect(function(dt)
            local ok, err = pcall(fn, dt)
        end)
    end

    -- ── Protected infinite loop (never dies from errors, respects interval) ──
    function lib.safeLoop(fn, interval)
        task.spawn(function()
            while true do
                local ok, err = pcall(fn)
                task.wait(interval or 1)
            end
        end)
    end

    -- ── SAB Place ID set (shared reference) ──
    lib.SAB_PLACE_IDS = {[109983668079237]=true,[85621847059032]=true,[99606176102979]=true}

    -- ── Teleport safety check (stop loops when game changes) ──
    function lib.isSABPlace()
        return lib.SAB_PLACE_IDS[game.PlaceId] == true
    end

    getgenv().__ASTRO_LIB = lib
end

local function pickWeightedAnimal()
    local pool = {}
    local totalWeight = 0
    for _, r in ipairs(RARITY_ORDER) do
        if trollSettings.enabledRarities[r] and ANIMALS_BY_RARITY[r] and #ANIMALS_BY_RARITY[r] > 0 then
            local w = RARITY_WEIGHTS[r] or 10
            totalWeight = totalWeight + w
            table.insert(pool, {rarity = r, weight = w})
        end
    end
    if totalWeight == 0 then return nil end
    local roll = math.random() * totalWeight
    local acc = 0
    local chosenRarity = pool[1].rarity
    for _, entry in ipairs(pool) do
        acc = acc + entry.weight
        if roll <= acc then chosenRarity = entry.rarity; break end
    end
    local animals = ANIMALS_BY_RARITY[chosenRarity]
    return animals[math.random(1, #animals)]
end
local function getRandomTraits(count)
    local all = {}
    if TraitsData then
        for t, _ in pairs(TraitsData) do
            if t ~= "Sleepy" then table.insert(all, t) end
        end
    end
    local picked = {}
    for i = #all, 2, -1 do
        local j = math.random(1, i)
        all[i], all[j] = all[j], all[i]
    end
    for i = 1, math.min(count, #all) do
        table.insert(picked, all[i])
    end
    return picked
end
local function getRandomMutation()
    local muts = {"Default"}
    if MutationsData then
        for m, _ in pairs(MutationsData) do table.insert(muts, m) end

end

-- Helper: Check if a plot has a real player owner currently in-game
local function plotHasActivePlayer(plotName, channel)
    -- Method 1: Check channel Owner data
    if channel then
        local ownerOk, ownerData = pcall(function() return channel:Get("Owner") end)
        if ownerOk and ownerData then
            local ownerName = nil
            if typeof(ownerData) == "Instance" and ownerData:IsA("Player") then
                ownerName = ownerData.Name
            elseif type(ownerData) == "table" and ownerData.Name then
                ownerName = ownerData.Name
            elseif type(ownerData) == "string" and ownerData ~= "" then
                ownerName = ownerData
            end
            if ownerName then
                -- Check if this player is actually in the server
                local player = Players:FindFirstChild(ownerName)
                if player then return true end
            end
        end
    end
    -- Method 2: Check plot's Owner attribute
    local plotObj = workspace:FindFirstChild("Plots") and workspace.Plots:FindFirstChild(plotName)
    if plotObj then
        local attrOwner = plotObj:GetAttribute("Owner")
        if attrOwner and attrOwner ~= "" then
            local player = Players:FindFirstChild(attrOwner)
            if player then return true end
        end
    end
    return false
end
local function trollScramblePlots()
    if trollScrambling then return 0, 0 end
    trollScrambling = true
    local plots = workspace:FindFirstChild("Plots")
    if not plots then
        print("[TrollScrambler] No Plots folder!")
        trollScrambling = false
        return 0, 0
    end
    local myPlot = getMyPlot()
    local myPlotName = myPlot and myPlot.Name or ""
    print("[TrollScrambler] My plot: " .. myPlotName)
    local channelMap = buildChannelMap()
    -- Fallback: cached channels
    if _channels2 then
        for key, ch in pairs(_channels2) do
            local idx = nil
            pcall(function() idx = ch:GetIndex() end)
            local mapKey = idx or key
            if type(mapKey) == "string" and not channelMap[mapKey] then channelMap[mapKey] = ch end
            if type(key) == "string" and not channelMap[key] then channelMap[key] = ch end
        end
    end
    local totalSpawned, totalSkipped, plotCount = 0, 0, 0
    local plotChildren = plots:GetChildren()
    print("[TrollScrambler] " .. #plotChildren .. " plot objects")
    for _, plot in ipairs(plotChildren) do
        if plot.Name ~= myPlotName then
            local channel = channelMap[plot.Name]
            if not channel then pcall(function() channel = _RawSync2:GetAllChannels()[plot.Name] end) end
            if not channel then pcall(function() channel = _channels2[plot.Name] end) end
            if channel then
                -- Skip plots without an active player (empty/unowned plots)
                if not plotHasActivePlayer(plot.Name, channel) then
                    print("[TrollScrambler] Skipping " .. plot.Name .. " (no active player)")
                else
                local podiumSet = {}
                local folder = plot:FindFirstChild("AnimalPodiums") or plot:FindFirstChild("Podiums")
                if folder then
                    for _, child in ipairs(folder:GetChildren()) do
                        local num = tonumber(child.Name)
                        if num then podiumSet[num] = true end
                    end
                end
                for i = 1, 10 do
                    local ok, data = pcall(function() return channel:Get("AnimalList." .. i) end)
                    if ok and data and type(data) == "table" then podiumSet[i] = true end
                end
                local podiums = {}
                for idx in pairs(podiumSet) do table.insert(podiums, idx) end
                table.sort(podiums)
                if #podiums > 0 then
                    plotCount = plotCount + 1
                    -- Backup original data for this plot before scrambling
                    if not backupData[plot.Name] then
                        backupData[plot.Name] = {}
                        for _, podIdx in ipairs(podiums) do
                            local ok, origData = pcall(function() return channel:Get("AnimalList." .. podIdx) end)
                            if ok then
                                backupData[plot.Name][podIdx] = origData
                            end
                        end
                    end
                    for _, podIdx in ipairs(podiums) do
                        if math.random(1, 100) > trollSettings.spawnChance then
                            totalSkipped = totalSkipped + 1
                        elseif trollSettings.emptyPodiumChance > 0 and math.random(1, 100) <= trollSettings.emptyPodiumChance then
                            pcall(function() channel:Set("AnimalList." .. podIdx, "Empty") end)
                            totalSkipped = totalSkipped + 1
                        else
                            local animal = pickWeightedAnimal()
                            if animal then
                                local mut = trollSettings.randomMutation and getRandomMutation() or "Default"
                                -- Weighted trait count
                                local traitCount = 0
                                if trollSettings.randomTraits then
                                    local roll = math.random(1, 100)
                                    if roll <= 30 then traitCount = 0
                                    elseif roll <= 55 then traitCount = 1
                                    elseif roll <= 75 then traitCount = 2
                                    elseif roll <= 88 then traitCount = 3
                                    elseif roll <= 96 then traitCount = 4
                                    else traitCount = 5 end
                                    traitCount = math.clamp(traitCount, trollSettings.minTraits, trollSettings.maxTraits)
                                end
                                local traits = traitCount > 0 and getRandomTraits(traitCount) or nil
                                local data = {
                                    Index = animal,
                                    Mutation = (mut ~= "Default" and mut ~= "Normal") and mut or nil,
                                    Traits = traits and #traits > 0 and traits or nil,
                                    LastCollect = workspace:GetServerTimeNow()
                                }
                                local ok, err = pcall(function() channel:Set("AnimalList." .. podIdx, data) end)
                                if ok then
                                    totalSpawned = totalSpawned + 1
                                else
                                    print("[TrollScrambler] Set FAILED " .. plot.Name .. " #" .. podIdx .. ": " .. tostring(err))
                                end
                            end
                        end
                    end
                end
                end -- end else (active player check)
            end
        end
    end
    print("[TrollScrambler] Done: " .. totalSpawned .. " spawned, " .. totalSkipped .. " skipped, " .. plotCount .. " plots")
    -- Mark scrambled plots in global registry
    for plotName, _ in pairs(backupData) do
        scrambledPlots[plotName] = true
    end
    hasBackup = true
    trollScrambling = false
    return totalSpawned, plotCount
end

-- ============================================================================
-- RESTORE FUNCTION — Reverse scramble using backed up data

-- ============================================================================
local function restoreScrambledPlots()
    if trollScrambling then return 0 end
    if not hasBackup or not next(backupData) then
        print("[TrollScrambler] No backup data to restore!")
        return 0
    end
    trollScrambling = true
    local channelMap = buildChannelMap()
    if _channels2 then
        for key, ch in pairs(_channels2) do
            local idx = nil
            pcall(function() idx = ch:GetIndex() end)
            local mapKey = idx or key
            if type(mapKey) == "string" and not channelMap[mapKey] then channelMap[mapKey] = ch end
            if type(key) == "string" and not channelMap[key] then channelMap[key] = ch end
        end
    end
    local restored, failed = 0, 0
    for plotName, podiums in pairs(backupData) do
        local channel = channelMap[plotName]
        if not channel then pcall(function() channel = _RawSync2:GetAllChannels()[plotName] end) end
        if not channel then pcall(function() channel = _channels2[plotName] end) end
        if channel then
            for podIdx, origData in pairs(podiums) do
                -- If original was nil/empty, set to "Empty" string (game format for empty podiums)
                local dataToRestore = origData
                if dataToRestore == nil then dataToRestore = "Empty" end
                local ok, err = pcall(function() channel:Set("AnimalList." .. podIdx, dataToRestore) end)
                if ok then
                    restored = restored + 1
                    print("[TrollScrambler] Restored " .. plotName .. " #" .. podIdx)
                else
                    failed = failed + 1
                    print("[TrollScrambler] FAILED restore " .. plotName .. " #" .. podIdx .. ": " .. tostring(err))
                end
            end
            -- Remove from fake registry
            scrambledPlots[plotName] = nil
        else
            print("[TrollScrambler] No channel found for " .. plotName .. " during restore!")
            failed = failed + 1
        end
    end
    backupData = {}
    hasBackup = false
    print("[TrollScrambler] Restored " .. restored .. " podiums! (" .. failed .. " failed)")
    trollScrambling = false
    return restored
end

-- ====== ICON CLICK → TOGGLE PANEL (hides icon when panel open) ======
iconBtn.MouseButton1Click:Connect(function()
    if iconHasDragged then return end -- Don't toggle if we just dragged
    local opening = not panel.Visible
    if opening then
        -- Clamp panel on-screen before showing
        local cam = workspace.CurrentCamera
        local vw = cam and cam.ViewportSize.X or 1920
        local vh = cam and cam.ViewportSize.Y or 1080
        local cx = math.clamp(panel.Position.X.Offset, 0, vw - PANEL_W)
        local cy = math.clamp(panel.Position.Y.Offset, 0, vh - PANEL_H)
        panel.Position = UDim2.new(0, cx, 0, cy)
    end
    panel.Visible = opening
    iconBtn.Visible = not opening -- Hide icon when panel is open
    Config.panelOpen = opening
    saveConfig()
end)

end
